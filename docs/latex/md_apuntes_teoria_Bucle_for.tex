\DoxyCodeLine{    \textcolor{comment}{// Código a repetir}}
\DoxyCodeLine{\}}

\end{DoxyCode}



\begin{DoxyItemize}
\item {\bfseries{Inicialización}}\+: Se ejecuta una vez al inicio. Suele declarar/inicializar la variable de control (ej\+: {\ttfamily int i = 0}).
\item {\bfseries{Condición}}\+: Se evalúa antes de cada iteración. Si es verdadera ({\ttfamily true}), el bucle continúa.
\item {\bfseries{Actualización}}\+: Se ejecuta al final de cada iteración (ej\+: incrementar/decrementar la variable).
\end{DoxyItemize}

\hypertarget{md_apuntes_teoria_Bucle_for_autotoc_md3}{}\doxysubsubsection{$<$strong$>$\+Consideraciones Importantes$<$/strong$>$}\label{md_apuntes_teoria_Bucle_for_autotoc_md3}

\begin{DoxyEnumerate}
\item {\bfseries{Variables de control}}\+:
\begin{DoxyItemize}
\item La variable de control puede declararse dentro del bucle (en C99 o superior)\+: 
\begin{DoxyCode}{0}
\DoxyCodeLine{\textcolor{keywordflow}{for} (\textcolor{keywordtype}{int} i = 0; i < 5; i++) \{ ... \}}

\end{DoxyCode}

\item En C antiguo, debe declararse fuera\+: 
\begin{DoxyCode}{0}
\DoxyCodeLine{\textcolor{keywordtype}{int} i;}
\DoxyCodeLine{\textcolor{keywordflow}{for} (i = 0; i < 5; i++) \{ ... \}}

\end{DoxyCode}

\end{DoxyItemize}
\item {\bfseries{Expresiones opcionales}}\+:
\begin{DoxyItemize}
\item Cualquier parte del {\ttfamily for} puede omitirse, pero los {\ttfamily ;} deben permanecer\+: 
\begin{DoxyCode}{0}
\DoxyCodeLine{\textcolor{keywordtype}{int} i = 0;}
\DoxyCodeLine{\textcolor{keywordflow}{for} (; i < 5; i++) \{ ... \} \textcolor{comment}{// Sin inicialización}}
\DoxyCodeLine{\textcolor{keywordflow}{for} (;;) \{ ... \}           \textcolor{comment}{// Bucle infinito (requiere `break` para salir)}}

\end{DoxyCode}

\end{DoxyItemize}
\item {\bfseries{Alcance de la variable}}\+:
\begin{DoxyItemize}
\item Si se declara dentro del {\ttfamily for}, solo existe dentro del bucle.
\end{DoxyItemize}
\item {\bfseries{Anidamiento}}\+:
\begin{DoxyItemize}
\item Los bucles {\ttfamily for} pueden anidarse para trabajar con estructuras multidimensionales.
\end{DoxyItemize}
\end{DoxyEnumerate}

\end{DoxyItemize}

Este bucle es ideal cuando se conoce el número de iteraciones de antemano. ¡\+Practica con variaciones para dominarlo!
